#include <cmath>
#include <iostream>
#include <string>
#include <vector>
using namespace std;

int N, K;
vector<vector<int>> cache;
vector<int> cacheOne;
vector<int> arr;
using ll = long long;

// 테이프의 길이가 length일때 최소로 사용되는 테이프의 갯수를 반환한다.
int fix(int length) {
  int countTape = 0;
  vector<bool> isMarking(arr.size(), false);
  for (int i = 0; i < arr.size(); i++) {
    if (!isMarking[i]) {
      int mark = arr[i] + length;
      countTape++;
      isMarking[i] = true;
      for (int j = i + 1; j < arr.size(); j++) {
        if (arr[j] < mark) {
          isMarking[j] = true;
        }
      }
    }
  }
  return countTape;
}

int main() {
  int n, l;
  std::cin >> n >> l;
  for (int i = 0; i < n; i++) {
    int temp;
    cin >> temp;
    arr.push_back(temp);
  }
  std::sort(arr.begin(), arr.end());
  std::cout << fix(l);
  return 0;
}
